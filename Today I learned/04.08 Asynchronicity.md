# Asynchronicity

```py

import asyncio

async def first_task():

    print("Task Starts")

    await asyncio.sleep(5)

    print("Task finishes")


async def main():

    print("Main Starts")

    task = asyncio.create_task(first_task())

    await asyncio.sleep(3)

    print("Main is running still")

    await task

    print("Main finishes")

asyncio.run(main())
```
